    const faqListEl = document.getElementById("faq-list"); const searchInputEl = document.getElementById("faq-search"); const noResultsEl = document.getElementById("no-results"); function createFaqItem(faq) { const li = document.createElement("li"); li.className = "faq-item"; li.innerHTML = `
    ${faq.answer}
    `; const btn = li.querySelector(".faq-question-btn"); const answer = li.querySelector(".faq-answer"); const toggle = li.querySelector(".faq-toggle"); btn.addEventListener("click", () => { const isOpen = li.classList.contains("open"); if (isOpen) { li.classList.remove("open"); answer.style.maxHeight = null; toggle.textContent = "+"; } else { li.classList.add("open"); answer.style.maxHeight = answer.scrollHeight + "px"; toggle.textContent = "−"; } }); return li; } function renderFaqs(filteredFaqs) { faqListEl.innerHTML = ""; filteredFaqs.forEach(faq => { faqListEl.appendChild(createFaqItem(faq)); }); noResultsEl.style.display = filteredFaqs.length ?
